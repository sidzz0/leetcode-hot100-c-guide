## 一、动态规划知识点回顾

DP 五步：

1. 定义 `dp` 含义
2. 写状态转移
3. 初始化
4. 确定遍历顺序
5. 举例验证

DP 常见信号：

- 求最值
- 求方案数
- 能否组成
- 当前状态依赖前面状态

## 1. 爬楼梯

### 思路

到第 i 阶可以从 i-1 或 i-2 来。

### C++ 代码

```cpp
class Solution {
public:
    int climbStairs(int n) {
        if (n <= 2) return n;
        int a = 1, b = 2;
        for (int i = 3; i <= n; i++) {
            int c = a + b;
            a = b;
            b = c;
        }
        return b;
    }
};
```

## 2. 杨辉三角

### 思路

每个位置等于上一行左上和右上的和。

### C++ 代码

```cpp
class Solution {
public:
    vector<vector<int>> generate(int numRows) {
        vector<vector<int>> ans(numRows);

        for (int i = 0; i < numRows; i++) {
            ans[i].assign(i + 1, 1);
            for (int j = 1; j < i; j++) {
                ans[i][j] = ans[i - 1][j - 1] + ans[i - 1][j];
            }
        }

        return ans;
    }
};
```

## 3. 打家劫舍

### 思路

到第 i 家，有两种选择：

- 不偷第 i 家：`dp[i-1]`
- 偷第 i 家：`dp[i-2] + nums[i]`

### C++ 代码

```cpp
class Solution {
public:
    int rob(vector<int>& nums) {
        int prev2 = 0, prev1 = 0;

        for (int x : nums) {
            int cur = max(prev1, prev2 + x);
            prev2 = prev1;
            prev1 = cur;
        }

        return prev1;
    }
};
```

## 4. 完全平方数

### 思路

`dp[i]` 表示组成 i 所需的最少完全平方数数量。

### C++ 代码

```cpp
class Solution {
public:
    int numSquares(int n) {
        vector<int> dp(n + 1, INT_MAX);
        dp[0] = 0;

        for (int i = 1; i <= n; i++) {
            for (int j = 1; j * j <= i; j++) {
                dp[i] = min(dp[i], dp[i - j * j] + 1);
            }
        }

        return dp[n];
    }
};
```

## 5. 零钱兑换

### 思路

完全背包。`dp[i]` 表示凑出金额 i 的最少硬币数。

### C++ 代码

```cpp
class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        vector<int> dp(amount + 1, amount + 1);
        dp[0] = 0;

        for (int i = 1; i <= amount; i++) {
            for (int coin : coins) {
                if (i >= coin) {
                    dp[i] = min(dp[i], dp[i - coin] + 1);
                }
            }
        }

        return dp[amount] > amount ? -1 : dp[amount];
    }
};
```

## 6. 单词拆分

### 思路

`dp[i]` 表示 `s[0..i)` 是否可以被拆分。枚举切割点 `j`，如果 `dp[j]` 为真且 `s[j..i)` 在字典中，则 `dp[i]` 为真。

### C++ 代码

```cpp
class Solution {
public:
    bool wordBreak(string s, vector<string>& wordDict) {
        unordered_set<string> words(wordDict.begin(), wordDict.end());
        vector<bool> dp(s.size() + 1, false);
        dp[0] = true;

        for (int i = 1; i <= s.size(); i++) {
            for (int j = 0; j < i; j++) {
                if (dp[j] && words.count(s.substr(j, i - j))) {
                    dp[i] = true;
                    break;
                }
            }
        }

        return dp[s.size()];
    }
};
```

## 7. 最长递增子序列

### 思路

`dp[i]` 表示以 `nums[i]` 结尾的最长递增子序列长度。

### C++ 代码

```cpp
class Solution {
public:
    int lengthOfLIS(vector<int>& nums) {
        int n = nums.size();
        vector<int> dp(n, 1);
        int ans = 1;

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < i; j++) {
                if (nums[j] < nums[i]) {
                    dp[i] = max(dp[i], dp[j] + 1);
                }
            }
            ans = max(ans, dp[i]);
        }

        return ans;
    }
};
```

## 8. 乘积最大子数组

### 思路

负数会让最大变最小、最小变最大，所以同时维护当前最大乘积和最小乘积。

### C++ 代码

```cpp
class Solution {
public:
    int maxProduct(vector<int>& nums) {
        int maxVal = nums[0];
        int minVal = nums[0];
        int ans = nums[0];

        for (int i = 1; i < nums.size(); i++) {
            int x = nums[i];
            if (x < 0) swap(maxVal, minVal);
            maxVal = max(x, maxVal * x);
            minVal = min(x, minVal * x);
            ans = max(ans, maxVal);
        }

        return ans;
    }
};
```

## 9. 分割等和子集

### 思路

如果总和是奇数，不可能分割。目标变成能否从数组中选一些数凑出 `sum / 2`，这是 01 背包。

### C++ 代码

```cpp
class Solution {
public:
    bool canPartition(vector<int>& nums) {
        int sum = accumulate(nums.begin(), nums.end(), 0);
        if (sum % 2) return false;

        int target = sum / 2;
        vector<bool> dp(target + 1, false);
        dp[0] = true;

        for (int x : nums) {
            for (int j = target; j >= x; j--) {
                dp[j] = dp[j] || dp[j - x];
            }
        }

        return dp[target];
    }
};
```

## 10. 最长有效括号

### 思路

`dp[i]` 表示以 `s[i]` 结尾的最长有效括号长度。只有 `s[i] == ')'` 时才可能形成有效括号。

如果前一个是 `(`：

```text
...()
```

如果前一个是 `)`，需要看前面是否有能匹配当前右括号的左括号。

### C++ 代码

```cpp
class Solution {
public:
    int longestValidParentheses(string s) {
        int n = s.size();
        vector<int> dp(n, 0);
        int ans = 0;

        for (int i = 1; i < n; i++) {
            if (s[i] == ')') {
                if (s[i - 1] == '(') {
                    dp[i] = (i >= 2 ? dp[i - 2] : 0) + 2;
                } else {
                    int j = i - dp[i - 1] - 1;
                    if (j >= 0 && s[j] == '(') {
                        dp[i] = dp[i - 1] + 2 + (j >= 1 ? dp[j - 1] : 0);
                    }
                }
                ans = max(ans, dp[i]);
            }
        }

        return ans;
    }
};
```
