## 1. 岛屿数量

### 思路

遇到一个陆地 `'1'`，答案加一，然后用 DFS/BFS 把与它连通的所有陆地淹掉，避免重复统计。

### C++ 代码

```cpp
class Solution {
public:
    int numIslands(vector<vector<char>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        int ans = 0;

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == '1') {
                    ans++;
                    dfs(grid, i, j);
                }
            }
        }

        return ans;
    }

    void dfs(vector<vector<char>>& grid, int i, int j) {
        int m = grid.size();
        int n = grid[0].size();
        if (i < 0 || i >= m || j < 0 || j >= n || grid[i][j] != '1') return;

        grid[i][j] = '0';
        dfs(grid, i + 1, j);
        dfs(grid, i - 1, j);
        dfs(grid, i, j + 1);
        dfs(grid, i, j - 1);
    }
};
```

复杂度：时间 O(mn)，空间 O(mn) 递归栈最坏情况。

## 2. 腐烂的橘子

### 思路

多源 BFS。初始所有烂橘子同时入队，每一轮让相邻新鲜橘子腐烂。层数就是分钟数。

### C++ 代码

```cpp
class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        queue<pair<int, int>> q;
        int fresh = 0;

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 2) q.push({i, j});
                if (grid[i][j] == 1) fresh++;
            }
        }

        int minutes = 0;
        int dirs[4][2] = {{1,0}, {-1,0}, {0,1}, {0,-1}};

        while (!q.empty() && fresh > 0) {
            int sz = q.size();
            minutes++;

            for (int k = 0; k < sz; k++) {
                auto [x, y] = q.front();
                q.pop();

                for (auto& d : dirs) {
                    int nx = x + d[0], ny = y + d[1];
                    if (nx < 0 || nx >= m || ny < 0 || ny >= n || grid[nx][ny] != 1) continue;
                    grid[nx][ny] = 2;
                    fresh--;
                    q.push({nx, ny});
                }
            }
        }

        return fresh == 0 ? minutes : -1;
    }
};
```

易错点：只有发生腐烂时才增加分钟，所以循环条件用 `fresh > 0`。

## 3. 课程表

### 思路

判断有向图是否有环。用拓扑排序：

- 建图
- 统计入度
- 入度为 0 的课程入队
- 每学完一门课，降低后继课程入度
- 最后如果学完课程数等于总数，则无环

### C++ 代码

```cpp
class Solution {
public:
    bool canFinish(int numCourses, vector<vector<int>>& prerequisites) {
        vector<vector<int>> graph(numCourses);
        vector<int> indegree(numCourses, 0);

        for (auto& p : prerequisites) {
            int a = p[0], b = p[1];
            graph[b].push_back(a);
            indegree[a]++;
        }

        queue<int> q;
        for (int i = 0; i < numCourses; i++) {
            if (indegree[i] == 0) q.push(i);
        }

        int count = 0;
        while (!q.empty()) {
            int cur = q.front();
            q.pop();
            count++;

            for (int nxt : graph[cur]) {
                indegree[nxt]--;
                if (indegree[nxt] == 0) q.push(nxt);
            }
        }

        return count == numCourses;
    }
};
```

## 4. 实现 Trie 前缀树

### 思路

Trie 是一棵字符树。每条边代表一个字符，从根到某个节点形成一个前缀。节点里的 `isEnd` 表示是否有单词在这里结束。

### C++ 代码

```cpp
class Trie {
private:
    struct Node {
        vector<Node*> children;
        bool isEnd;
        Node() : children(26, nullptr), isEnd(false) {}
    };

    Node* root;

public:
    Trie() {
        root = new Node();
    }
    
    void insert(string word) {
        Node* cur = root;
        for (char c : word) {
            int idx = c - 'a';
            if (!cur->children[idx]) {
                cur->children[idx] = new Node();
            }
            cur = cur->children[idx];
        }
        cur->isEnd = true;
    }
    
    bool search(string word) {
        Node* node = find(word);
        return node && node->isEnd;
    }
    
    bool startsWith(string prefix) {
        return find(prefix) != nullptr;
    }

private:
    Node* find(const string& s) {
        Node* cur = root;
        for (char c : s) {
            int idx = c - 'a';
            if (!cur->children[idx]) return nullptr;
            cur = cur->children[idx];
        }
        return cur;
    }
};
```
